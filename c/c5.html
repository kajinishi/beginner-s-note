◯戻り値
関数で処理した結果を返すときの値
return 戻り値; として使う。returnをするとそこで関数が終了する
戻り値を持たない関数のときは戻り値の型をvoidにする

◯引数
関数に引き渡す値
      
        #include<stdio.h>
        int sum(int a,int b){//戻り値をint型に指定
          return a+b;//a+bを戻り値として返す
        }
        void show(int a,int b,int n){//戻り値がvoid(何も返さない)
          printf("%d + %d = %d\n",a,b,n);
          return;//関数はreturnで終了する
          puts("わたしは嘘つきです。さて、正直者でしょうか嘘つきでしょうか");
        }
        int main(){
          int a=10,b=30;
          int n=sum(a,b);
          show(a,b,n);
        }
      
10 + 30 = 40

引数の渡し方

        #include<stdio.h>
        void func(int a){
          a+=10;
          printf("func関数の変数a = %d\n",a);
        }
        int main(){
         int a=10;
         func(a);
         printf("main関数の変数a = %d\n",a);
        }
      
func関数の変数a = 20
main関数の変数a = 10
      
        #include<stdio.h>
        int func(int a){
          a+=10;
          printf("func関数の変数a = %d\n",a);
          return a;
        }
        int main(){
         int a=10;
         a=func(a);
         printf("main関数の変数a = %d\n",a);
        }
      
func関数の変数a = 20
main関数の変数a = 20
      
        #include<stdio.h>
        void func(int *a,int *b){
          *a=100;
          int i;
          for(i=0;i<3;i++)b[i]=i*10;
          return;
        }
        int main(){
         int a=10;
         int b[3]={0};//これで配列の値が全部0になる
         func(&a,b);
         printf("a = %d\n",a);
         int i;
         for(i=0;i<3;i++){
           printf("b[%d] = %d\n",i,b[i]);
         }
        }
      
a = 100
b[0] = 0
b[1] = 10
b[2] = 20

        }
        int main(){

◯関数ポインタの宣言
戻り値の型 (*ポインタ名) (引数型 引数名の並び)
      
        #include<stdio.h>
        void add(int a,int b);
        void sub(int a,int b);
        void mul(int a,int b);
        void div(int a,int b);
        int main(){
          int x=10,y=2;
          int i;
          void (*func[])(int,int)={add,sub,mul,div};
          for(i=0;i<4;i++){
            func[i](x,y);
          }
          return 0;
        }
        void add(int a,int b){
          printf("a %c b = %d\n",'+',a+b);
        }
        void sub(int a,int b){
          printf("a %c b = %d\n",'-',a-b);
        }
        void mul(int a,int b){
          printf("a %c b = %d\n",'*',a*b);
        }
        void div(int a,int b){
          printf("a %c b = %d\n",'/',a/b);
        }
      
a + b = 12
a - b = 8
a * b = 20
a / b = 5
