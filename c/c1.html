書式指定

#include<stdio.h>
int main(){
 printf("%sで%dを表示\n","書式指定",3);
 return 0;
}

実行結果

書式指定で3を表示
%d 整数を10進数で表示
%f 実数(小数点あるやつ)を表示
%c 文字(' 'で囲まれた半角文字1文字)を表示
%s 文字列(" "で囲まれた文字)を表示

変数

整数型

型名 範囲 サイズ（ビット数）
char -128 ~ 127 8
short -32768 ~ 32767 16
int システムで異なる -
long -2147483648 ~ 2147483647 32

実数型

型名 範囲 サイズ（ビット数）
float -3.4×10の38乗 ~ 3.4×10の38乗 32
double -1.7×10の308 ~ 1.7×10の308乗 64
参考：cの絵本
#include<stdio.h>
int main(){
 int a;//変数の宣言は 変数型　変数名
 a=3;  //値を代入
 double b=1.5,c=3.14;/* 宣言と同時に代入も可能 */
 printf("a=%d,b=%f,c=%f\n",a,b,c);
 return 0;
}

実行結果

a=3,b=1.500000,c=3.140000

演算

演算子 意味 使い方
+ 足す a=b+c;(aにはbとcを足した値が入る)
- 引く a=b-c;(aにはbからcを引いた値が入る)
* かける a=b*c;(aにはbにcをかけた値が入る)
/ 割る a=b/c;(aにはbをcで割った値が入る)
% 余り a=b%c;(aにはbをcで割った余りが入る)
= 代入 a=a+b;(aにはもともとaに代入されていた値にbを足した値が入る)
演算子 使い方
+= a+=b;(a=a+bと同じ)
-= a-=b;(a=a-bと同じ)
*= a*=b;(a=a*bと同じ)
/= a/=b;(a=a/bと同じ)
%= a%=b;(a=a%bと同じ) 
          #include<stdio.h>
          int main(){
            int a=3,b=5,c=7;
            a=b+3;
            b=c-b;
            c/=2;//c=c/2
            printf("b+3=%d,c-b=%d\n",a,b );
            printf("a*b=%d,c/2=%d",a*b,c);
            return 0;
          }

実行結果

b+3=8,b-c=2
a*b=16,c/2=3
      
#include<stdio.h>

  printf("実数/整数：(5.0)/3=%f\n", 5.0/3);
  printf("実数/実数：(5.0)/(3.0)=%f\n",f);

  return 0;
}

実行結果

整数/整数：5/3=1.000000
整数/実数：5/(3.0)=1.666667
実数/整数：(5.0)/3=1.666667
実数/実数：(5.0)/(3.0)=1.666667
演算子 使い方
++ a++または++a(a=a+1と同じ)
-- a--または--a(a=a-1と同じ) 
          #include<stdio.h>
          int main(){
            int x=0,y=0;
            int a=1,b=1;
            x=++a;//前置
            y=b++;//後置
            printf("前置だと足した後に代入なのでx=%dだけど\n後置だと代入した後bを足すのでy=%d\n",x,y);
            printf("出力のとき前置だとまず引いて%d,後置だとまだひかずに%dになる\n",--a,b--);
            printf("でも結局a=%d,b=%dで最後はおなじ\n",a,b);
            return 0;
          }

実行結果

前置だと足した後に代入なのでx=2だけど
後置だと代入した後bを足すのでy=1
出力のとき前置だとまず引いて1,後置だとまだひかずに2になる
でも結局a=1,b=1で最後はおなじ
